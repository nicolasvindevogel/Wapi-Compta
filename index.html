-- WAPI COMPTA V5 -- Lots, identites (occupants), repartition keys/items, and extra metadata for CODA/lots. begin; alter table if exists public.compta_lots add column if not exists reference text, add column if not exists address text, add column if not exists city text, add column if not exists floor text, add column if not exists notes text, add column if not exists active boolean not null default true; alter table if exists public.compta_owners add column if not exists iban text; alter table if exists public.compta_suppliers add column if not exists phone text, add column if not exists address text, add column if not exists iban text; create table if not exists public.compta_occupants ( id uuid primary key default gen_random_uuid(), copro_id uuid not null references public.compta_copros(id) on delete cascade, lot_id uuid null references public.compta_lots(id) on delete set null, display_name text not null, email text null, phone text null, address text null, iban text null, active boolean not null default true, notes text null, created_by uuid null, created_at timestamptz not null default now(), updated_at timestamptz not null default now() ); create index if not exists idx_compta_occupants_copro_id on public.compta_occupants(copro_id); create index if not exists idx_compta_occupants_lot_id on public.compta_occupants(lot_id); create table if not exists public.compta_distribution_keys ( id uuid primary key default gen_random_uuid(), copro_id uuid not null references public.compta_copros(id) on delete cascade, name text not null, is_default boolean not null default false, active boolean not null default true, created_by uuid null, created_at timestamptz not null default now(), updated_at timestamptz not null default now() ); create unique index if not exists ux_compta_distribution_keys_default on public.compta_distribution_keys(copro_id, is_default) where is_default = true; create table if not exists public.compta_distribution_items ( id uuid primary key default gen_random_uuid(), distribution_key_id uuid not null references public.compta_distribution_keys(id) on delete cascade, lot_id uuid not null references public.compta_lots(id) on delete cascade, included boolean not null default true, quotities numeric(14,4) not null default 0, created_by uuid null, created_at timestamptz not null default now(), updated_at timestamptz not null default now(), unique(distribution_key_id, lot_id) ); create index if not exists idx_compta_distribution_items_key on public.compta_distribution_items(distribution_key_id); create index if not exists idx_compta_distribution_items_lot on public.compta_distribution_items(lot_id); -- Seed default key "Quotites generales" for existing copros if absent. insert into public.compta_distribution_keys (copro_id, name, is_default) select c.id, 'Quotites generales', true from public.compta_copros c where not exists ( select 1 from public.compta_distribution_keys k where k.copro_id = c.id and k.is_default = true ); -- Seed default distribution items from current lots/quotities when missing. insert into public.compta_distribution_items (distribution_key_id, lot_id, included, quotities) select k.id, l.id, true, coalesce(l.quotities, 0) from public.compta_lots l join public.compta_distribution_keys k on k.copro_id = l.copro_id and k.is_default = true where not exists ( select 1 from public.compta_distribution_items i where i.distribution_key_id = k.id and i.lot_id = l.id ); commit;
